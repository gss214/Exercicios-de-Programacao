import java.util.Scanner;

public class ExerciciosURI {
    public static void main(String[] args) {
        Scanner t = new Scanner (System.in);
        
        int n = t.nextInt();
      
        for (int i = 1; i <= n; i++) {
            if (n % i == 0){
                System.out.println(i);
            }
        }
    }
}
